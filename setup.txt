Set Up the project using vite and dependency

npm create vite@latest 
 -project name - client //(or any name your choice)
 -fremwork - react
 -varient - javascript

cd client  // replace client with your project name
npm install

npm install react-router-dom  // For routing
npm install lucide-react   // For Icons
npm install react-hot-toast //For popup message
npm install react-player   //To play the videos

npm run dev //to start the project
